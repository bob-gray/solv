    • .shim(name:string, implementation:function) returns function

      Attaches an instance method only if it doesn't already exist Returns self to allow for chaining

      name
      Name of method
    • .shimStatic(name:string, implementation:function) returns function

      Attaches a static method only if it doesn't already exist Returns self to allow for chaining

      name
      If name is not passed it will be retrieved with implementation.getName()

	Function.prototype.shim = shim;
	Function.prototype.shimStatic = shimStatic;
	
	function shim (name, implementation) {
		attachIfUndefined(this.prototype, name, implementation);

		return this;
	}
	
	function shimStatic (name, implementation) {
		attachIfUndefined(this, name, implementation);

		return this;
	}
	
	function attachIfUndefined (target, name, implementation) {
		if (!target[name]) {
			target[name] = implementation;
		}
	}
});
